const originalFinishRound = Game.prototype.finishRound;
Game.prototype.finishRound = function patchedFinishRound() {
  const shouldFinish = this.state === 'playing';
  originalFinishRound.call(this);

  const roundPoints = Math.max(0, Math.floor(this.player.score - this.roundScoreStart));
  dom.roundReport.textContent = `${this.roundCleaned} itens recolhidos • ${this.roundReports || 0} denúncias • ${roundPoints.toLocaleString()} pontos • ${this.roundSewageHits} contatos com esgoto`;

};

const originalCreateSewagePatch = Game.prototype.createSewagePatch;
Game.prototype.createSewagePatch = function patchedCreateSewagePatch(...args) {
  originalCreateSewagePatch.apply(this, args);
  const item = this.pollution[this.pollution.length - 1];
  if (!item || item.kind !== 'sewage') return;

  item.reported = false;
  const beacon = new THREE.Group();
  const baseMaterial = toonMaterial(0x111417);
  const redMaterial = toonMaterial(0xff3b30, { emissive: 0x8a0700, emissiveIntensity: 0.7 });
  const whiteMaterial = toonMaterial(0xffffff);

  const float = new THREE.Mesh(new THREE.CylinderGeometry(0.34, 0.46, 0.3, 10), baseMaterial);
  float.position.y = 0.12;
  const pole = new THREE.Mesh(new THREE.CylinderGeometry(0.055, 0.075, 1.25, 8), baseMaterial);
  pole.position.y = 0.82;
  const lamp = new THREE.Mesh(new THREE.SphereGeometry(0.24, 10, 7), redMaterial);
  lamp.position.y = 1.52;
  lamp.scale.y = 0.78;
  const exclamation = new THREE.Mesh(new THREE.BoxGeometry(0.08, 0.25, 0.08), whiteMaterial);
  exclamation.position.set(0, 1.55, 0.22);
  const dot = new THREE.Mesh(new THREE.SphereGeometry(0.055, 7, 5), whiteMaterial);
  dot.position.set(0, 1.36, 0.22);
  beacon.add(float, pole, lamp, exclamation, dot);
  beacon.position.set(item.radius + 1.45, 0.08, 0);
  beacon.userData.lamp = lamp;
  item.mesh.add(beacon);
  item.beacon = beacon;
};

Game.prototype.reportSewage = function reportSewage(item) {
  if (!item || item.reported) return;
  item.reported = true;
  this.player.reported = (this.player.reported || 0) + 1;
  this.roundReports = (this.roundReports || 0) + 1;
  const points = Math.floor(260 * this.player.multiplier);
  this.player.score += points;
  this.player.flow = clamp(this.player.flow + 10, 0, 100);
  this.cleanupCharge += 1;

  if (item.beacon?.userData.lamp) {
    item.beacon.userData.lamp.material.color.setHex(0x75f0a1);
    item.beacon.userData.lamp.material.emissive?.setHex(0x167a3b);
  }

  const position = item.mesh.position.clone();
  position.y += 1.4;
  this.createBurst(position, [0xff3b30, 0xffffff, 0x75f0a1, 0xffef64], 15);
  this.showCallout(`DENÚNCIA + AÇÃO +${points}`);
  this.audio.tone(430, 0.08, 'square', 0.07, 360);
  setTimeout(() => this.audio.tone(760, 0.12, 'triangle', 0.07, 260), 70);
  navigator.vibrate?.([18, 35, 24]);
  this.cameraShake = Math.max(this.cameraShake, 0.08);
  this.squash = Math.max(this.squash, 0.16);

  if (this.cleanupCharge >= this.mods.cleanupThreshold) {
    this.cleanupCharge -= this.mods.cleanupThreshold;
    this.grantCleanupReward();
  }
  this.updateHud();
};

const originalUpdateObjects = Game.prototype.updateObjects;
Game.prototype.updateObjects = function patchedUpdateObjects(dt) {
  const wantsReport = this.input.pressed('report');
  originalUpdateObjects.call(this, dt);

  let nearest = null;
  let nearestDistance = Infinity;
  for (const item of this.pollution) {
    if (item.kind !== 'sewage' || item.reported) continue;
    const dx = item.mesh.position.x - this.player.x;
    const dz = item.mesh.position.z - this.playerZ;
    const distance = Math.hypot(dx, dz);
    const outerRange = item.radius + 3.1;
    const safeInnerRange = item.radius * 0.68;

    if (item.beacon) {
      const pulse = 1 + Math.sin(this.time * 7 + item.phase) * 0.12;
      item.beacon.scale.setScalar(pulse);
      item.beacon.rotation.y = -item.mesh.rotation.y;
    }

    if (distance < outerRange && distance > safeInnerRange && distance < nearestDistance) {
      nearest = item;
      nearestDistance = distance;
    }
  }

  const available = Boolean(nearest && this.state === 'playing');
  dom.reportPrompt?.classList.toggle('show', available);
  dom.reportButton?.classList.toggle('available', available);
  dom.reportButton?.setAttribute('aria-disabled', available ? 'false' : 'true');
  if (available && wantsReport) this.reportSewage(nearest);
};

const originalCollectTrash = Game.prototype.collectTrash;
Game.prototype.collectTrash = function patchedCollectTrash(...args) {
  const previous = this.player.cleaned;
  originalCollectTrash.apply(this, args);
  if (this.player.cleaned > previous && this.player.cleaned % 10 === 0) {
    this.showCallout('TRANSFORME INDIGNAÇÃO EM ATITUDE');
  }
};

const originalGrantCleanupReward = Game.prototype.grantCleanupReward;
Game.prototype.grantCleanupReward = function patchedGrantCleanupReward() {
  originalGrantCleanupReward.call(this);
  if (dom.callout?.textContent.startsWith('ECO BONUS')) {
    dom.callout.textContent = dom.callout.textContent.replace('ECO BONUS', 'AÇÃO AMBIENTAL');
  }
};
