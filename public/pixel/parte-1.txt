window.addEventListener('keydown', (evento) => {
  const acao = MAPA_TECLAS[evento.code];
  if (!acao) return;
  evento.preventDefault();
  if (!teclas.has(acao)) apertou.add(acao);
  teclas.add(acao);
});
window.addEventListener('keyup', (evento) => {
  const acao = MAPA_TECLAS[evento.code];
  if (!acao) return;
  evento.preventDefault();
  teclas.delete(acao);
});
window.addEventListener('blur', () => teclas.clear());

function segurando(acao) { return teclas.has(acao); }
function consumiu(acao) {
  if (!apertou.has(acao)) return false;
  apertou.delete(acao);
  return true;
}

for (const botao of document.querySelectorAll('[data-acao]')) {
  const acao = botao.dataset.acao;
  const ativar = (evento) => {
    evento.preventDefault();
    if (!teclas.has(acao)) apertou.add(acao);
    teclas.add(acao);
    botao.classList.add('ativo');
    botao.setPointerCapture?.(evento.pointerId);
    navigator.vibrate?.(8);
  };
  const desativar = (evento) => {
    evento.preventDefault();
    teclas.delete(acao);
    botao.classList.remove('ativo');
  };
  botao.addEventListener('pointerdown', ativar);
  botao.addEventListener('pointerup', desativar);
  botao.addEventListener('pointercancel', desativar);
}

class Som8Bits {
  constructor() {
    this.contexto = null;
    this.ligado = true;
  }
  iniciar() {
    if (!this.contexto) this.contexto = new (window.AudioContext || window.webkitAudioContext)();
    this.contexto?.resume();
  }
  tom(frequencia = 440, duracao = .08, tipo = 'square', volume = .035, deslize = 0) {
    if (!this.ligado || !this.contexto) return;
    const agora = this.contexto.currentTime;
    const oscilador = this.contexto.createOscillator();
    const ganho = this.contexto.createGain();
    oscilador.type = tipo;
    oscilador.frequency.setValueAtTime(frequencia, agora);
    if (deslize) oscilador.frequency.linearRampToValueAtTime(Math.max(40, frequencia + deslize), agora + duracao);
    ganho.gain.setValueAtTime(volume, agora);
    ganho.gain.exponentialRampToValueAtTime(.0001, agora + duracao);
    oscilador.connect(ganho).connect(this.contexto.destination);
    oscilador.start(agora);
    oscilador.stop(agora + duracao);
  }
  coleta() { this.tom(680, .06, 'square', .035, 300); }
  pulo() { this.tom(190, .12, 'triangle', .05, 240); }
  impacto() { this.tom(90, .22, 'sawtooth', .055, -25); }
  premio() { this.tom(420, .09, 'square', .04, 480); setTimeout(() => this.tom(760, .08, 'triangle', .03, 180), 75); }
}

function inteiro(n) { return Math.round(n); }
function limitar(v, a, b) { return Math.max(a, Math.min(b, v)); }
function aleatorio(a, b) { return a + Math.random() * (b - a); }
function escolher(lista) { return lista[Math.floor(Math.random() * lista.length)]; }
function embaralhar(lista) { return [...lista].sort(() => Math.random() - .5); }
function colisao(a, b, raio) { return Math.hypot(a.x - b.x, a.y - b.y) < raio; }

function ret(x, y, w, h, cor) {
  ctx.fillStyle = cor;
  ctx.fillRect(inteiro(x), inteiro(y), inteiro(w), inteiro(h));
}

function texto(txt, x, y, cor = COR.claro, tamanho = 8, alinhar = 'left') {
  ctx.fillStyle = cor;
  ctx.font = `900 ${tamanho}px "Courier New", monospace`;
  ctx.textAlign = alinhar;
  ctx.textBaseline = 'top';
  ctx.fillText(txt, inteiro(x), inteiro(y));
}

function caixa(x, y, w, h, cor = COR.noite, borda = COR.claro) {
  ret(x, y, w, h, COR.preto);
  ret(x + 2, y + 2, w - 4, h - 4, borda);
  ret(x + 4, y + 4, w - 8, h - 8, cor);
}

function quebrarTexto(txt, largura) {
  const palavras = txt.split(' ');
  const linhas = [];
  let atual = '';
  for (const palavra of palavras) {
    const teste = atual ? `${atual} ${palavra}` : palavra;
    if (teste.length > largura && atual) {
      linhas.push(atual);
      atual = palavra;
    } else atual = teste;
  }
  if (atual) linhas.push(atual);
  return linhas;
}

function ondaY(x, fase) {
  const parede = x > 250 ? -(x - 250) * .45 : 0;
  return 116
    + Math.sin((x + fase) * .045) * 6
    + Math.sin((x * .12) + fase * .06) * 2.5
    + parede;
}
